        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);

        // משתני מערכת
        let user = null;
        let myData = { name: '', score: 0, completedStages: [] };
        let allQuestions = [];
        let gameSettings = { stage1: true, stage2: false, stage3: false, stage4: false };
        let currentStage = null;
        let currentQIndex = 0;
        let timer = 20;
        let timerId = null;

        // הפעלה ראשונית
        async function init() {
            signInAnonymously(auth);
            onAuthStateChanged(auth, async (u) => {
                if (u) {
                    user = u;
                    setupListeners();
                    checkSavedUser();
                }
            });
        }

        // בדיקת משתמש קיים
        async function checkSavedUser() {
            const savedName = localStorage.getItem('quiz_name');
            if (savedName) {
                const userDoc = await getDoc(doc(db, "players", user.uid));
                if (userDoc.exists()) {
                    myData = userDoc.data();
                    updateLobbyUI();
                    showScreen('lobby');
                }
            }
        }

        function setupListeners() {
            // האזנה להגדרות שלבים
            onSnapshot(doc(db, "settings", "game_state"), (snap) => {
                if (snap.exists()) {
                    gameSettings = snap.data();
                    renderStages();
                    renderAdminToggles();
                }
            });

            // האזנה לשאלות
            onSnapshot(doc(db, "content", "questions"), (snap) => {
                if (snap.exists()) {
                    allQuestions = snap.data().list;
                    renderAdminQuestions();
                }
            });

            // האזנה לניקוד אישי (לעדכונים)
            onSnapshot(doc(db, "players", user.uid), (snap) => {
                if (snap.exists()) {
                    myData = snap.data();
                    updateLobbyUI();
                }
            });
        }

        // פונקציות מסכים
        window.showScreen = (id) => {
            document.querySelectorAll('body > div').forEach(d => d.classList.add('hidden'));
            document.getElementById(`screen-${id}`).classList.remove('hidden');
        };

        window.joinGame = async () => {
            const name = document.getElementById('player-name-input').value.trim();
            if (!name) return alert("נא להזין שם");
            myData = { name, score: 0, completedStages: [] };
            await setDoc(doc(db, "players", user.uid), myData);
            localStorage.setItem('quiz_name', name);
            showScreen('lobby');
        };

        function updateLobbyUI() {
            document.getElementById('display-name').innerText = myData.name;
            document.getElementById('display-score').innerText = myData.score;
        }

        function renderStages() {
            const grid = document.getElementById('stages-grid');
            grid.innerHTML = '';
            for (let i = 1; i <= 4; i++) {
                const isOpen = gameSettings[`stage${i}`];
                const isDone = myData.completedStages.includes(i);
                const btn = document.createElement('button');
                btn.className = `stage-card p-8 rounded-2xl border-2 transition font-bold text-xl flex flex-col items-center ${
                    isDone ? 'bg-emerald-900/30 border-emerald-500 text-emerald-300' :
                    isOpen ? 'bg-indigo-900/30 border-indigo-500 hover:bg-indigo-900/50' : 'bg-slate-800 border-slate-700 opacity-50 cursor-not-allowed'
                }`;
                btn.innerHTML = `<span>שלב ${i}</span><span class="text-sm mt-2">${isDone ? '✅ הושלם' : isOpen ? '🔓 פתוח' : '🔒 נעול'}</span>`;
                if (isOpen && !isDone) btn.onclick = () => startStage(i);
                grid.appendChild(btn);
            }
        }

        window.startStage = (stageNum) => {
            currentStage = stageNum;
            currentQIndex = 0;
            showScreen('game');
            loadQuestion();
        };

        function loadQuestion() {
            clearInterval(timerId);
            const stageQs = allQuestions.filter(q => q.stage == currentStage);
            if (!stageQs[currentQIndex]) return finishStage();

            const q = stageQs[currentQIndex];
            document.getElementById('q-progress').innerText = `שאלה ${currentQIndex + 1} מתוך ${stageQs.length}`;
            document.getElementById('question-text').innerText = q.text;
            
            const grid = document.getElementById('options-grid');
            grid.innerHTML = '';
            const colors = ['border-rose-500/40 text-rose-200', 'border-blue-500/40 text-blue-200', 'border-amber-500/40 text-amber-200', 'border-emerald-500/40 text-emerald-200'];
            
            q.options.forEach((opt, i) => {
                const b = document.createElement('button');
                b.className = `answer-btn p-6 rounded-2xl bg-slate-800 border-2 text-xl font-bold transition hover:bg-slate-700 ${colors[i]}`;
                b.innerText = opt;
                b.onclick = () => checkAnswer(i, q.correct);
                grid.appendChild(b);
            });

            timer = 20;
            updateTimerUI();
            timerId = setInterval(() => {
                timer--;
                updateTimerUI();
                if (timer <= 0) checkAnswer(-1, q.correct);
            }, 1000);
        }

        function updateTimerUI() {
            document.getElementById('timer-num').innerText = timer;
            document.getElementById('timer-bar').style.width = `${(timer / 20) * 100}%`;
        }

        async function checkAnswer(idx, correct) {
            clearInterval(timerId);
            const isCorrect = idx === correct;
            const feedback = document.getElementById('feedback-msg');
            const inner = feedback.querySelector('div');
            
            if (isCorrect) {
                const points = 1000 + (timer * 50);
                myData.score += points;
                inner.innerText = `+${points}`;
                inner.className = "text-6xl font-black px-10 py-5 rounded-2xl shadow-2xl animate-bounce bg-emerald-600 text-white";
                await updateDoc(doc(db, "players", user.uid), { score: myData.score });
            } else {
                inner.innerText = idx === -1 ? "נגמר הזמן!" : "טעות!";
                inner.className = "text-6xl font-black px-10 py-5 rounded-2xl shadow-2xl animate-bounce bg-rose-600 text-white";
            }

            feedback.classList.remove('hidden');
            setTimeout(() => {
                feedback.classList.add('hidden');
                showLeaderboard();
            }, 2000);
        }

        async function showLeaderboard() {
            const snap = await getDocs(query(collection(db, "players"), orderBy("score", "desc"), limit(5)));
            const body = document.getElementById('leaderboard-body');
            body.innerHTML = '';
            let i = 1;
            snap.forEach(d => {
                const p = d.data();
                body.innerHTML += `<tr class="${d.id === user.uid ? 'bg-indigo-900/30' : ''}">
                    <td class="p-4 font-bold">${i === 1 ? '🥇' : i === 2 ? '🥈' : i === 3 ? '🥉' : i}</td>
                    <td class="p-4">${p.name}</td>
                    <td class="p-4 font-mono text-indigo-400 font-bold">${p.score}</td>
                </tr>`;
                i++;
            });
            showScreen('leaderboard');
        }

        window.nextStep = () => {
            currentQIndex++;
            const stageQs = allQuestions.filter(q => q.stage == currentStage);
            if (currentQIndex < stageQs.length) loadQuestion();
            else finishStage();
        };

        async function finishStage() {
            if (!myData.completedStages.includes(currentStage)) {
                myData.completedStages.push(currentStage);
                await updateDoc(doc(playersPath(), user.uid), { completedStages: myData.completedStages });
            }
            alert("כל הכבוד! סיימת את השלב.");
            showScreen('lobby');
        }

        window.loginAdmin = () => {
            const pass = document.getElementById('admin-pass').value;
            if (pass === "1234" || pass === "admin123") {
                showScreen('adminDashboard');
            } else alert("סיסמה שגויה");
        };

        function renderAdminToggles() {
            const cont = document.getElementById('admin-toggles');
            cont.innerHTML = '';
            for (let i = 1; i <= 4; i++) {
                const active = gameSettings[`stage${i}`];
                cont.innerHTML += `<div class="flex justify-between items-center p-3 bg-slate-700 rounded-xl">
                    <span>שלב ${i}</span>
                    <input type="checkbox" ${active ? 'checked' : ''} onchange="toggleStage(${i}, this.checked)" class="w-6 h-6 accent-indigo-500">
                </div>`;
            }
        }

        window.toggleStage = async (num, val) => {
            gameSettings[`stage${num}`] = val;
            await setDoc(doc(db, "settings", "game_state"), gameSettings);
        };

        function renderAdminQuestions() {
            const cont = document.getElementById('admin-questions');
            cont.innerHTML = '';
            allQuestions.forEach((q, i) => {
                cont.innerHTML += `<div class="bg-slate-700 p-4 rounded-xl border border-slate-600">
                    <div class="text-xs text-indigo-300 mb-1">שלב ${q.stage} | שאלה ${i+1}</div>
                    <input value="${q.text}" onchange="updateQ(${i}, 'text', this.value)" class="w-full bg-slate-800 p-2 rounded border border-slate-600 mb-2">
                    <div class="grid grid-cols-2 gap-2">
                        ${q.options.map((opt, oi) => `<input value="${opt}" onchange="updateQOpt(${i}, ${oi}, this.value)" class="bg-slate-800 p-2 rounded text-sm border border-slate-600">`).join('')}
                    </div>
                </div>`;
            });
        }

        window.updateQ = async (i, field, val) => {
            allQuestions[i][field] = val;
            await setDoc(doc(db, "content", "questions"), { list: allQuestions });
        };

        window.updateQOpt = async (i, oi, val) => {
            allQuestions[i].options[oi] = val;
            await setDoc(doc(db, "content", "questions"), { list: allQuestions });
        };

        window.resetAllScores = async () => {
            if (!confirm("לאפס תוצאות לכולם?")) return;
            const snap = await getDocs(collection(db, "players"));
            snap.forEach(async d => await updateDoc(doc(db, "players", d.id), { score: 0, completedStages: [] }));
            alert("אופס בהצלחה");
        };

        init();
